import requests
import json

# GET-запрос

url = "https://jsonplaceholder.typicode.com/posts"

response = requests.get(url)
posts = response.json()
posts_users = [post for post in posts if post['userId'] % 2 == 0]

for post in posts_users:
    print(post)

#POST-запрос

url = "https://jsonplaceholder.typicode.com/posts"

post_data = {
    "title": "Тестовый пост",
    "body": "Тело тестового поста"
}

response = requests.post(url, json=post_data)
post = response.json()

print(json.dumps(post, ensure_ascii=False, indent=4))

#PUT-запрос

post_id = 1 #101 не работает, так как фактически на сайте запрос не создать
url = f"https://jsonplaceholder.typicode.com/posts/{post_id}"

post_data_new = {
    "title": "Обновлённый пост",
    "body": "Тело обновлённого поста",
    "id": post_id
}

response = requests.put(url, json=post_data_new)

print(json.dumps(post, ensure_ascii=False, indent=4))
